CREATE TABLE `AccountLegal` (
  `LP_ID` int(11) NOT NULL AUTO_INCREMENT,
  `Hotel_ID` int(11) DEFAULT NULL,
  `CS_ID` int(11) DEFAULT NULL,
  `Balance` int(11) DEFAULT NULL,
  PRIMARY KEY (`LP_ID`),
  KEY `CS_ID_idx` (`CS_ID`),
  CONSTRAINT `CS_ID` FOREIGN KEY (`CS_ID`) REFERENCES `cashsystem` (`CS_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8
