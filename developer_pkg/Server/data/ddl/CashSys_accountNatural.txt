CREATE TABLE `AccountNatural` (
  `NP_ID` int(11) NOT NULL AUTO_INCREMENT,
  `CS_ID` int(11) NOT NULL,
  `Name` varchar(45) DEFAULT NULL,
  `Email` varchar(45) DEFAULT NULL,
  `Balance` int(11) DEFAULT NULL,
  PRIMARY KEY (`NP_ID`,`CS_ID`),
  KEY `CS_ID_fk1_idx` (`CS_ID`),
  CONSTRAINT `CS_ID_fk1` FOREIGN KEY (`CS_ID`) REFERENCES `cashsystem` (`CS_ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=7 DEFAULT CHARSET=utf8
